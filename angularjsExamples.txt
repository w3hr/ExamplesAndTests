
	
	//Simple App + Controller creation
	
	<div ng-app="myApp" ng-controller="myController">
	{{foo}}
	</div>


	<script>
		var app = angular.module('myApp', []);

		app.controller('myController', function ($scope) {
			$scope.foo = 'bar';
		}
	</script>
	
	
	//Simple Watch
	$scope.$watch('foobar', function() {
		console.warn('foobar has changed!');
	});
	
	
	//angular forEach
	
	let found = false;
	angular.forEach($scope.list, function (value, key) {
		if (key === 'foo')
			found = value;
	});
	
	if (!found)
		console.log('found foo: ' + found);
	else
		console.warn('didnt found foo :( !');
